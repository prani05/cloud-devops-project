pipeline {
    agent any

    stages {
        stage('Checkout Code') {
            steps {
                git 'https://github.com/your-repo/database-backup.git'
            }
        }

        stage('Run Ansible Playbook') {
            steps {
                sh 'ansible-playbook -i hosts.ini backup.yml'
            }
        }
    }

    post {
        success {
            echo '✅ Database backup completed successfully!'
        }
        failure {
            echo '❌ Backup failed! Check logs for details.'
        }
    }
}
